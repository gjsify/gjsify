<Code code={`import Adw from 'gi://Adw?version=1';
import Gtk from 'gi://Gtk?version=4.0';
import { CanvasWebGLWidget } from '@gjsify/webgl';
import * as THREE from 'three';
import { TeapotGeometry } from 'three/addons/geometries/TeapotGeometry.js';

const app = new Adw.Application({ application_id: 'gjsify.examples.teapot' });
app.connect('activate', () => {
    const glArea = new CanvasWebGLWidget();
    glArea.onReady((canvas) => {
        const scene = new THREE.Scene();
        const teapot = new THREE.Mesh(new TeapotGeometry(300, 15));
        scene.add(teapot);
    });
});`} lang="ts" title="teapot.ts" />
